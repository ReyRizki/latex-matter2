    4.a \\
    $$\lim_{t \to 2} \biggr[\frac{t^2 + 6t - 12}{t - 2} \, \hat i + \frac{t + \sqrt 2}{t^2 - 2} \, \hat j\biggr]$$
    $$= \lim_{t \to 2} \biggr[\frac{(t + 8)(t - 2)}{t - 2} \, \hat i + \frac{t + \sqrt 2}{(t + \sqrt 2)(t - \sqrt 2)} \, \hat j\biggr]$$
    $$= \lim_{t \to 2} \biggr[t + 8 \, \hat i + \frac{1}{t - \sqrt 2} \, \hat j\biggr]$$
    $$= 10 \, \hat i + \frac{1}{2 - \sqrt 2} \, \hat j$$

    4.b \\
    $$\lim_{t \to \pi} \biggr[\frac{2 \cos 2t}{\cos^2t} \, \hat i + \frac{2 \cos 2t}{\sin^2 (2\pi - t)} \, \hat j\biggr]$$
    $$= \frac{2}{1} \, \hat i + \frac{2}{1} \, \hat j$$
    $$= 2 \, \hat i + 2 \, \hat j$$

    4.c \\
    $$\lim_{(x, y) \to (2, 3)} \frac{3x^2y}{3x + 2y} + \frac{2x^2y}{8y - 4}$$
    $$= \frac{36}{12} + \frac{24}{20}$$
    $$= 3 + \frac{6}{5}$$
    $$= 4 \frac{1}{5}$$

    4.d \\
    $$\lim_{(x, y) \to (\pi, 0)} \frac{\sin y}{\cos x}$$
    $$= \lim_{(x, y) \to (\pi, 0)} \frac{0}{- 1}$$
    $$= 0$$
